```python
def solution(people, limit):
    end = 0
    people.sort()
    interval_sum = 0
    cnt = 0
    
    for start in range(len(people)): 
        while interval_sum <= limit and end < len(people):
            interval_sum += people[end]
            end += 1
            
        if interval_sum > limit:
            end -= 1
            interval_sum -= people[end]
            interval_sum -= people[start]
        cnt += 1
    return cnt 
```
