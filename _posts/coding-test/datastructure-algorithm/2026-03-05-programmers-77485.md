# [[2021 Dev-Matching: 웹 백엔드 개발자(상반기)] 행렬 테두리 회전하기](https://school.programmers.co.kr/learn/courses/30/lessons/77485)


# ✅ 코드

```java
class Solution {
    static int[] dr = {0,1,0,-1};
    static int[] dc = {1,0,-1,0};
    
    
    public int[] solution(int rows, int columns, int[][] queries) {
        int[] answer = new int[queries.length];
        
        // 1. 초기 배열 생성
        int[][] arr = new int[rows][columns];
        int num = 1;
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < columns; c++) {
                arr[r][c] = num++;
            }
        }
            
        
        // 요청문 갯수
        int T = queries.length;
        for (int t = 0; t < T; t++) {
            int sr = queries[t][0];
            int sc = queries[t][1];
            int er = queries[t][2];
            int ec = queries[t][3];
            int row = (er-sr+1);
            int col = (ec-sc+1);
            int N;
            if (row-2 == 0 || col - 2 == 0) N = row * col;
            else N = row * col - ( (row-2) * (col-2) );
            
            // 2-1. 회전 배열
            int[] rotate_arr = new int[N];
            // 2-2. 좌표 저장할 배열
            int[][] pos = new int[N][2];
            sr--;
            sc--;
            er--;
            ec--;
            
            // 3. 탐색 시작전 1번째 좌표 삽입하고, 저장후에 시작
            int start = 0;
            rotate_arr[start] = arr[sr][sc];
            pos[start][0] = sr;
            pos[start][1] = sc;
            start++;
            
            // 4. 탐색 시작
            int cr = sr;
            int cc = sc;
            int d = 0;
            while (start < N) {
                int nr = cr + dr[d];
                int nc = cc + dc[d];
                // 범위 초과시 방향 전환
                if (nr < sr || nr > er || nc < sc || nc > ec) {
                    d = (d+1) % 4;
                    continue;
                }
                // 범위밖이 아니라면
                rotate_arr[start] = arr[nr][nc];
                pos[start][0] = nr;
                pos[start][1] = nc;
                // 현재좌표 최신화
                cr = nr;
                cc = nc;
                start++;
            } 
            
            
            // 5. 1회전
            for (int i = 0; i < pos.length; i++) {
                int cur_val = rotate_arr[i];
                int r = pos[ (i+1) % pos.length ][0];
                int c = pos[ (i+1) % pos.length ][1];
                arr[r][c] = cur_val;
            }
            
            // 6. 최소값 찾기
            int min = Integer.MAX_VALUE;
            for (int i = 0; i < rotate_arr.length; i++) {
                if (rotate_arr[i] < min) {
                    min = rotate_arr[i];
                }
            }
            
            // 7. 최소값 추가 후에 다음으로 전환
            answer[t] = min;
        }
        
        return answer;
    }
}
```
