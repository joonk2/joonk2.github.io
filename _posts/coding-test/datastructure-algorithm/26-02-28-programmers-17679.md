class Solution {
    public int solution(int m, int n, String[] board) {

        int row = m;
        int col = n;

        while (true) {
            boolean[][] visited = new boolean[row][col];
            boolean can_delete_block = false;
            
            for (int r = 0; r < row-1; r++) {
                for (int c = 0; c < col-1; c++) {

                        visited[r][c+1] = true;
                        visited[r+1][c+1] = true;
                        visited[r+1][c] = true;

                for (int c = 0; c < col; c++) {

            }
